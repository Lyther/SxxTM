								NOT = { exists = owner }

						}

						limit = {

							is_under_colonization = no
							is_occupied_flag = no
							OR = {
								NOT = { exists = owner }
								owner = { is_same_value = root }
							}
						}
						if = {
							limit = {
								NOT = { exists = owner }
							}
							set_owner = root
						}
						save_event_target_as = sxx_heaven

					limit = {

							is_planet_class = pc_sxx_heaven
						}
						is_occupied_flag = no
					}
					every_owned_pop = {
						limit = {
							has_job = sxx_other_fairy
						}
