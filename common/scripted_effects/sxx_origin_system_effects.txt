sxx_add_task_counter = {
    if = {
        limit = {
            has_event_chain = sxx_task_$TASK$_chain
        }
        add_event_chain_counter = {
            event_chain = sxx_task_$TASK$_chain
            counter = sxx_task_$TASK$_counter
            amount = 1
        }
        if = {
            limit = {
                has_completed_event_chain_counter = {
                    event_chain = sxx_task_$TASK$_chain
                    counter = sxx_task_$TASK$_counter
                }
            }
            country_event = { id = sxx_task_finish.$TASK$ }
        }
    }
}

sxx_task_month_pulse_effect = {
    if = {
        limit = {
            has_event_chain = sxx_task_$TASK$_chain
        }
        add_event_chain_counter = {
            event_chain = sxx_task_$TASK$_chain
            counter = sxx_time_remaining
            amount = 1
        }
        if = {
            limit = {
                has_completed_event_chain_counter = {
                    event_chain = sxx_task_$TASK$_chain
                    counter = sxx_time_remaining
                }
            }
            country_event = { id = sxx_task_fail.$TASK$ }
        }
        else = {
            country_event = { id = sxx_task_month_pulse.$TASK$ days = 30 }
        }
    }
}
