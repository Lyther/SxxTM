# Add cultivator pop flag
country_event = {
	id = sxxtm_event.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_ai = no
		is_sxx_sect = yes
	}

	immediate = {
		if = {
			limit = {
				is_sxx_sect = yes
				NOT = { has_event_chain = sxx_cultivation_data_chain }
			}
			country_event = { id = sxx_origin.1 }
		}
		every_owned_pop = {
			limit = {
				NOT = { has_pop_flag = sxx_cultivator }
			}
			set_pop_flag = sxx_cultivator
		}
		country_event = { id = sxxtm_event.2 days = 1 }
	}
}
