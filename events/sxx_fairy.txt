	id = sxx_fairy.150
	hide_window = yes

	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				is_sxx_heaven_country = yes
			}
			every_owned_planet = {
				limit = {
					is_planet_class = pc_sxx_heaven
				}
				set_variable = {
					which = sxx_heaven_stability_value
					value = value:sxx_get_heaven_stability_value
				}
	
				if = {
					limit = {
						NOT = {
							has_building = building_sxx_lingxiao
						}
					}
					add_building = building_sxx_lingxiao
				}

		}

			limit = {
				has_country_flag = sxx_country_get_old_zhenxian

country_event = {
	id = sxx_fairy.170
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		is_sxx_heaven_country = yes
		has_edict = sxx_heaven_pop_resettle
		any_system_within_border = {
			any_system_planet = {
				is_planet_class = pc_sxx_heaven
				is_under_colonization = no
				is_occupied_flag = no
				OR = {
					NOT = { exists = owner }
					owner = { is_same_value = root }
				}
			}
		}
		#any_owned_planet = {
		#	is_planet_class = pc_sxx_heaven
		#	is_under_colonization = no
		#	is_occupied_flag = no
		#}
	}

	immediate = {
		random_system_within_border = {
			limit = {
				any_system_planet = {
					is_planet_class = pc_sxx_heaven
					is_under_colonization = no
					is_occupied_flag = no
					OR = {
						NOT = { exists = owner }
						owner = { is_same_value = root }
					}
				}
			}
			random_system_planet = {
				limit = {
					is_planet_class = pc_sxx_heaven
					is_under_colonization = no
					is_occupied_flag = no
					OR = {
						NOT = { exists = owner }
						owner = { is_same_value = root }
					}
				}
				if = {
					limit = {
						NOT = { exists = owner }
					}
					set_owner = root
				}
				save_event_target_as = sxx_heaven
			}
		}
		#random_owned_planet = {
		#	limit = {
		#		is_planet_class = pc_sxx_heaven
		#		is_under_colonization = no
		#		is_occupied_flag = no
		#	}
		#	save_event_target_as = sxx_heaven
		#}
		every_owned_planet = {
			limit = {
				NOT = {
					is_planet_class = pc_sxx_heaven
				}
				is_occupied_flag = no
			}
			every_owned_pop = {
				limit = {
					has_job = sxx_other_fairy
				}
				resettle_pop = {
					pop = this
					planet = event_target:sxx_heaven
				}
			}
		}
	}
}
