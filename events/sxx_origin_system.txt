country_event = {
	id = sxx_origin_system.1
	title = sxx_origin_system_name
	desc = sxx_coins_show
	is_triggered_only = yes

	diplomatic = yes

	picture_event_data = {
		room = no_video_feed_room
	}

	trigger = {
		has_origin = origin_sxx_system
	}

	option = {
		name = sxx_origin_system.1.a
		hidden_effect = { country_event = { id = sxx_origin_system.100 } }
	}
	option = {
		name = sxx_origin_system.1.b
		allow = {
			if = {
				limit = {
					check_variable = { which = sxx_task_num value >= sxx_task_max }
				}
				custom_tooltip = {
					fail_text = sxx_task_num_is_max
					always = no
				}
			}
			else_if = {
				limit = {
					has_country_flag = sxx_getting_task
				}
				custom_tooltip = {
					fail_text = sxx_is_getting_task
					always = no
				}
			}
			else = {
				hidden_trigger = { always = yes }
			}
		}
		hidden_effect = {
			set_timed_country_flag = { flag = sxx_getting_task days = 360 }
			country_event = { id = sxx_origin_system.200 }
		}
	}
	option = {
		name = sxx_system_close
		default_hide_option = yes
	}
}
